import axios from 'axios';

const data = { username: 'John', password: 'Doe' };
const config = {
  headers: {
    'Content-Type': 'application/json'
  }
};

axios.post('https://api.example.com/login', data, config)
  .then(response => {
    console.log(response.data);
  })
  .catch(error => {
    console.error(error);
  });
